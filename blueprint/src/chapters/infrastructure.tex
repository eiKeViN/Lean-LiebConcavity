\section{OpStar Isomorphism}

\begin{proposition}[CFC via the opposite-algebra isomorphism]
  \label{prop:opstar_cfc}
  \lean{opStar_map_cfc, op_rpow_eq_rpow_op_nonneg, op_rpow_eq_rpow_op_strictlyPositive}
  \leanok
  Let $A$ be an ordered unital C$^*$-algebra.  The map
  $\varphi : A \to A^{\mathrm{op}}$, $a \mapsto a^*$, is a star algebra
  isomorphism, and it intertwines the continuous functional calculus:
  for any continuous $f$ and self-adjoint $a \in A$,
  \[
    \varphi(f(a)) = f(\varphi(a)).
  \]
  In particular, for $a \geq 0$ and $r \geq 0$, or for $a$ strictly positive and any
  $r \in \mathbb{R}$:
  \[
    (a^r)^{\mathrm{op}} = (a^{\mathrm{op}})^r.
  \]
  \begin{proof}
    \proves{prop:opstar_cfc}
    \leanok
    $\varphi$ is a star algebra homomorphism, so the CFC intertwining property applies
    directly.
  \end{proof}
\end{proposition}

\section{Unitary Conjugation}

\begin{theorem}[CFC commutes with unitary conjugation]
  \label{thm:cfc_unitary_conj}
  \lean{CStarAlgebra.cfc_unitary_conj'}
  \leanok
  Let $A$ be a unital C$^*$-algebra, $u \in A$ unitary, and $f : \mathbb{R} \to \mathbb{R}$
  continuous.  For any self-adjoint $a \in A$,
  \[
    f(u^* a\, u) = u^*\, f(a)\, u.
  \]
  \begin{proof}
    \proves{thm:cfc_unitary_conj}
    \leanok
    Unitary conjugation $x \mapsto u^* x u$ is a star algebra automorphism, so the
    CFC intertwining property gives $f(u^* a u) = u^* f(a) u$.
  \end{proof}
\end{theorem}

\section{Diagonal CFC}

\begin{theorem}[CFC distributes over diagonal matrices]
  \label{thm:cfc_diagonal}
  \lean{MatCStar.cfc_diagonal}
  \leanok
  Let $d_0,\ldots,d_{n-1} \in \mathbb{R}$ and $f : \mathbb{R} \to \mathbb{R}$
  continuous.  Then
  \[
    f\!\bigl(\operatorname{diagonal}(d_0,\ldots,d_{n-1})\bigr)
    \;=\;
    \operatorname{diagonal}(f(d_0),\ldots,f(d_{n-1})),
  \]
  where the CFC is taken in $M_n(\mathbb{C})$.
  \begin{proof}
    \proves{thm:cfc_diagonal}
    \leanok
    Diagonal matrices in $M_n(\mathbb{C})$ form a commutative C$^*$-subalgebra
    isomorphic to $\mathbb{C}^n$, and the CFC on a diagonal matrix acts entry-wise.
  \end{proof}
\end{theorem}

\begin{theorem}[Order on diagonal matrices is entry-wise]
  \label{thm:diagonal_le_diagonal}
  \lean{MatCStar.diagonal_le_diagonal_iff}
  \leanok
  For $d_0,\ldots,d_{n-1},\, e_0,\ldots,e_{n-1} \in \mathbb{R}$,
  \[
    \operatorname{diagonal}(d_0,\ldots,d_{n-1}) \leq \operatorname{diagonal}(e_0,\ldots,e_{n-1})
    \;\iff\;
    \forall i,\; d_i \leq e_i,
  \]
  where the order on $M_n(\mathbb{C})$ is the Loewner (positive semidefinite) order.
  \begin{proof}
    \proves{thm:diagonal_le_diagonal}
    \leanok
    A diagonal matrix is positive semidefinite iff all diagonal entries are nonneg,
    and the difference of diagonal matrices is diagonal, so the characterization follows.
  \end{proof}
\end{theorem}

\section{Reindex as Star Algebra Equivalence}

\begin{proposition}[Reindex as a star algebra equivalence]
  \label{prop:reindex_star_alg}
  \lean{Matrix.reindexStarAlgEquiv}
  \leanok
  For finite index sets $m$, $n$ of equal cardinality, a bijection $e : m \to n$,
  and a star algebra $A$, the map $M_m(A) \to M_n(A)$ given by
  $(M_{ij}) \mapsto (M_{e^{-1}(i),\,e^{-1}(j)})$ is a star algebra equivalence.
  \begin{proof}
    \proves{prop:reindex_star_alg}
    \leanok
    Direct verification: reindexing preserves ring operations, star, and scalar action.
  \end{proof}
\end{proposition}
